[docs]    def genere_son(self):
        """ Genere le signal sonore de base """    
        harmoniques = [\
#            {'gain':1.0, 'freq':50},
            {'gain':1.0, 'freq':1500},
#            {'gain':2.0, 'freq':1700},
         #   {'gain':1.0, 'freq':1800},
            {'gain':2.0, 'freq':3000},
            #    {'gain':2.0, 'freq':10000},
           # {'gain':1.5, 'freq':11900},
            {'gain':3.0, 'freq':12000},
           # {'gain':1.0, 'freq':13050},
            ]

        x = 0.0* np.random.randn(len(self.temps))
        for ha in harmoniques:
           x = x + ha.get('gain') * np.sin(2. * np.pi * ha.get('freq') * self.temps)
        son = x / np.max(np.abs(x))
        return son

[docs]    def calcul_module_fft(self,signal):
        """ Calcul du module de la FFT de signal"""
        spectref = np.fft.fft(signal)
        spectref = 20*np.log10(np.abs(spectref))
        return spectref

        """ Fonction encodage en 32 sous bandes """
        
        padding = np.zeros(32 - input_audio.shape[0]%32)
        input_audio = np.concatenate([input_audio, padding])
        
        
        X= np.zeros(512)
        plt.figure()

        for k in range (0,len(input_audio),32):
#            for i in np.arange(511,31,-1):
#                X[i]=X[i-32]
            X[:31:-1]=X[-33::-1]
            for i in np.arange(31,-1,-1):
                 X[i]=input_audio[k-i]
            
            Signal_encode=self.calcul_sous_bandes(X,k)

        return Signal_encode

        """ Calcul des 32 sous bandes """
        #Constitution de la liste des composantes tonales        
        comp_ton=self.tonal_comp (X,k)
        
        Z= np.zeros(512)    
        #construction des coeficients Zi    
        for i in range(512):        
                Z[i]=filtre.C[i]* X[i]      
        
        #construction des coeficients Yi 
        Y= np.zeros(64) 
        for z in range(64):
            for j in range(8):
                Y[z] += Z[z+64*j]
             
        #construction des Sous bandes de sortie  
        Sortie= np.zeros(32)   
        for i in range(32):
            for k in range(64):
#                Sortie[i] += Y[k] * np.cos((2*i+1)*(k-16)*np.pi/64.0)
                Sortie[i] += Y[k] * self.M[i,k]

        return Sortie

[docs]    def decode(self,Sub_band):
        """ Fonction de decodage a partir des 32 sous bandes de frequence"""
        V= np.zeros(1024)
        Echant_decode=np.zeros(0)    
        for z in range(12*32):
#            for i in np.arange(1023,63,-1):
#                V[i]=V[i-64]
            V[:63:-1]=V[-65::-1]
            V[0:64] = 0.0
            for i in range(64):
               for k in range(32):
                   V[i] += Sub_band[k] * np.cos((2*k+1)*(i+16)*np.pi/64)
#                   V[i] += Sub_band[k] * self.N[k,i]
            U=np.zeros(512)
            for i in range(8):
                for j in range(32):
                    U[(i*64)+j]=V[(i*128)+j]
                    U[(i*64)+32+j]=V[(i*128)+(96+j)]
                
            W=np.zeros(512)            
            for i in range(512):
                W[i]=U[i]*filtre.D[i]
                
            S=np.zeros(32)    
            for j in range(32):
               for i in range(16):
                   b=31-j
                   S[b]=S[b]+W[j+32*i]
                   

            Echant_decode = np.concatenate([S,Echant_decode])
            
        plt.figure()
        plt.title('Decodage')
        plt.xlabel('Temps (s)')
        plt.ylabel('Amplitude (V)')
        plt.plot(Echant_decode[10*32*32:])

        return Echant_decode

        '''
                
        
        Detection des composantes tonales du son
        
        :param son_buffer: Je sait plus mais c'est un test
        :type son_buffer: integer or None  
        
        
        
        
        
        
        .. math::
            \sum_{a}^{b} x_n = c_1 + c_2^3 + \cos(\pi)
            
        
        
        .. warning:: Du calme hein vraiment c'est juste un test !!
        
        .. note:: Ceci est une boite « Note »
        
        .. seealso:: Ceci est une boite « voir aussi »
        
        .. rubric:: Footnotes 

        .. [#f1] Ceci est un test
        .. [#f2] Oui oui un test
        .. [#f3] Non mis c'est vraiment un test hein ^^
        
        '''
        son_bufferf=son_buffer*np.hanning(512)
        X= self.calcul_module_fft(son_bufferf)
        X_tonal=[]
        for i in np.arange(2,251):
            if X[i+1]<=X[i] and X[i-1]<X[i]:
                if i<63:
                    j_valeurs=[-2,2]
                elif i<127:
                    j_valeurs=[-3,-2,2,3]
                else:
                    j_valeurs=[-6,-5,-4,-3,-2,2,3,4,5,6]        
                if np.array([X[i]-X[i+j]>=7 for j in j_valeurs]).all():    
                   X_tonal.append(i)
        if k<=1300:
            plt.gca().cla()
            plt.title('Composantes tonales')
            plt.xlabel('Frequence (Hz)')
            plt.ylabel('Gain (dB)')
            fft_size=len(X)
            f=np.arange(0,0.5*self.fs,self.fs/fft_size)
            plt.plot(f,X[:len(f)])
            for tonal in X_tonal:
                plt.plot((tonal*self.fs)/fft_size,X[tonal],'ro')
            plt.pause(0.001)
        return X
